# Problem 2390: Removing Stars From a String

```python
class Solution:
  def removeStars(self, s: str) -> str:
    ans = []
    for c in s:
      if c == '*':
        ans.pop()
      else:
        ans.append(c)
    return ''.join(ans)
```

class Solution {
  public String removeStars(String s) {
    StringBuilder sb = new StringBuilder();
    for (final char c : s.toCharArray())
      if (c == '*')
        sb.deleteCharAt(sb.length() - 1);
      else
        sb.append(c);
    return sb.toString();
  }
}
