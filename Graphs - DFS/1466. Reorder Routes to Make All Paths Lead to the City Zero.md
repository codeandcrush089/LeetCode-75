# Problem 1466: Reorder Routes to Make All Paths Lead to the City Zero

## ðŸ Python Solution
```python
from collections import deque

class Solution:
    def minReorder(self, n: int, connections: list[list[int]]) -> int:
        # adj[node] = [(neighbor, cost), ...]
        # cost 1 means the edge points AWAY from 0 (needs flip)
        # cost 0 means the edge points TOWARD 0 (is correct)
        adj = [[] for _ in range(n)]
        for u, v in connections:
            adj[u].append((v, 1)) # Original direction: u -> v
            adj[v].append((u, 0)) # Reverse direction: v -> u (free)

        total_reorders = 0
        visited = {0}
        queue = deque([0])

        while queue:
            curr = queue.popleft()
            for neighbor, cost in adj[curr]:
                if neighbor not in visited:
                    visited.add(neighbor)
                    # If we traverse an original edge (cost 1) while 
                    # moving away from 0, it means that edge needs to be flipped
                    total_reorders += cost
                    queue.append(neighbor)
                    
        return total_reorders
```

class Solution {
 public:
  int minReorder(int n, vector<vector<int>>& connections) {
    vector<vector<int>> graph(n);

    for (const vector<int>& connection : connections) {
      const int u = connection[0];
      const int v = connection[1];
      graph[u].push_back(v);
      graph[v].push_back(-u);  // - := u -> v
    }

    return dfs(graph, 0, -1);
  }

 private:
  int dfs(const vector<vector<int>>& graph, int u, int prev) {
    int change = 0;

    for (const int v : graph[u]) {
      if (abs(v) == prev)
        continue;
      if (v > 0)
        ++change;
      change += dfs(graph, abs(v), u);
    }

    return change;
  }
};

class Solution {
  public int minReorder(int n, int[][] connections) {
    List<Integer>[] graph = new List[n];
    Arrays.setAll(graph, i -> new ArrayList<>());

    for (int[] connection : connections) {
      final int u = connection[0];
      final int v = connection[1];
      graph[u].add(v);
      graph[v].add(-u); // - := u -> v
    }

    return dfs(graph, 0, -1);
  }

  private int dfs(List<Integer>[] graph, int u, int prev) {
    int change = 0;

    for (final int v : graph[u]) {
      if (Math.abs(v) == prev)
        continue;
      if (v > 0)
        ++change;
      change += dfs(graph, Math.abs(v), u);
    }

    return change;
  }
}

## ðŸ”¹ Approach Used

* **Graph Traversal with BFS (Edge Reorientation Counting)**
* Convert the directed graph into an undirected graph with edge costs:

  * Cost `1` if the edge is originally directed **away from city 0** (needs reordering).
  * Cost `0` if the edge is directed **toward city 0** (already correct).
* Start BFS from city `0` and visit all reachable cities.
* While traversing, add the edge cost whenever moving along an edge that needs to be reversed.
* The total accumulated cost gives the minimum number of reorders.
