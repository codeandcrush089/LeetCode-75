class Solution:
  def kidsWithCandies(
      self,
      candies: list[int],
      extraCandies: int,
  ) -> list[bool]:
    maxCandy = max(candies)
    return [candy + extraCandies >= maxCandy for candy in candies]

class Solution {
  public List<Boolean> kidsWithCandies(int[] candies, int extraCandies) {
    List<Boolean> ans = new ArrayList<>();
    final int maxCandy = Arrays.stream(candies).max().getAsInt();

    for (final int candy : candies)
      ans.add(candy + extraCandies >= maxCandy);

    return ans;
  }
}

## ğŸ”¹ Logic 

* First, find the **maximum number of candies** any kid currently has.
* For each kid:

  * Add `extraCandies` to their current candies.
  * Check if the new total is **greater than or equal to** the maximum value.
* If yes, mark `True`; otherwise, mark `False`.
* Return the list of boolean results for all kids.
