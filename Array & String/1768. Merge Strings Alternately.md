class Solution {
 public:
  string mergeAlternately(string word1, string word2) {
    const int n = min(word1.length(), word2.length());
    string prefix;

    for (int i = 0; i < n; ++i) {
      prefix += word1[i];
      prefix += word2[i];
    }

    return prefix + word1.substr(n) + word2.substr(n);
  }
};

class Solution {
  public String mergeAlternately(String word1, String word2) {
    final int n = Math.min(word1.length(), word2.length());
    StringBuilder sb = new StringBuilder();

    for (int i = 0; i < n; ++i) {
      sb.append(word1.charAt(i));
      sb.append(word2.charAt(i));
    }

    return sb.append(word1.substring(n)).append(word2.substring(n)).toString();
  }
}

## ðŸ”¹ Approach Used

* **Two-Pointer / Parallel Traversal Approach**
* Traverse both strings together character by character.
* Append one character from `word1`, then one from `word2`.
* If one string ends earlier, continue appending characters from the remaining string.
